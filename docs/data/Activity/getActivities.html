static Map<String, Activity> getActivities(Map json) {
	final Map<String, Activity> result = {};
	for (final MapEntry entry in json.entries) {
		result [entry.key] = Activity.fromJson(
			Map.from(entry.value)
		);
	}
	return result;
}
